AsRef<T>


#![allow(unused)]
fn main() {
fn is_hello<T: AsRef<str>>(s: T) {
   assert_eq!("hello", s.as_ref());
}
let s = "hello";
is_hello(s);
}

as_ref() 不能出错, 否则就需要用 Option or a Result<T,E> 形式



enmu match 组合


#![allow(unused)]
fn main() {
enum Message {
    Move(Point),
    Echo(String),
    ChangeColor((u8, u8, u8)),
    Quit,
}
fn process(&mut self, message: Message) {
    match message {
        Message::Move(value) => {
            self.move_position(value) 
        }
        Message::ChangeColor(value) => {
            self.change_color(value) 
        }
        Message::Echo(value) =>{
            self.echo(value)
        }
        Message::Quit =>{self.quit()}
    }
}
}

其中value是枚举成员中包含的任意类型值, 如果枚举成员中有2个以上变量可用 ( _, _, value) 形式提取任意变量
